    let currentLang = 'lv'; /* sākuma valoda latviešu */

    function setLang(lang) { /* funkcija, kas maina valodu */
      currentLang = lang; /* iestata pašreizējo valodu */
      document.getElementById('title').innerText = translations[lang].title; /* maina virsrakstu */
      document.getElementById('content').innerText = translations[lang].content; /* maina saturu */

      document.querySelectorAll('.lang-option').forEach(el => { /* cikls pa valodu pogām */
        el.classList.remove('active'); /* noņem aktīvo klasi */
        if (el.dataset.lang === lang) el.classList.add('active'); /* iestata aktīvo pogu */
      });
    }

    function toggleTheme() { /* funkcija tēmas maiņai */
      document.body.classList.toggle('dark-theme'); /* pievieno/ņem tumšo tēmu */
    }

    function renderLangOptions() { /* izveido valodu izvēles pogas */
      const langSelector = document.getElementById('langSelector'); /* atrod bloku */
      langSelector.innerHTML = ''; /* iztīra saturu */
      for (let code in translations) { /* cikls pa valodu kodiem */
        const langDiv = document.createElement('div'); /* izveido jaunu elementu */
        langDiv.className = 'lang-option'; /* piešķir klasi */
        langDiv.dataset.lang = code; /* piešķir valodas kodu */
        langDiv.innerHTML = `
          <img src="https://flagcdn.com/w40/${code === 'en' ? 'gb' : code}.png" alt="${translations[code].name}" />
          ${translations[code].name}
        `; /* pievieno karogu un valodas nosaukumu */
        langDiv.onclick = () => setLang(code); /* klikšķa notikums valodas maiņai */
        if (code === currentLang) langDiv.classList.add('active'); /* aktivē izvēlēto valodu */
        langSelector.appendChild(langDiv); /* pievieno pogu lapā */
      }
    }

    renderLangOptions(); /* ielādē valodu pogas */
